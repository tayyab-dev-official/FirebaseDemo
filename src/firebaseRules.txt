rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function isUserLoggedIn() {
        return request.auth != null;
    }

    function isUserAuthorOfPost(){
        return request.auth.uid == resource.data.uid;
    }

    // Deny all read and write access requests.
    match /{document=**} {
      allow read, write: if false;
    }

    // All authenticated users 
    match /{document=**} {
      allow read, write: if request.auth != null;
    }

    // All posts of logged-in user only
    match /{document=**} {
        allow write: if isUserLoggedIn();
        allow read: if isUserLoggedIn() && isUserAuthorOfPost();
    }

    // Allow only authenticated content owners access
    match /some_collection/{userId}/{document} {
      allow read, write: if request.auth != null && request.auth.uid == userId
    }
  }
}
